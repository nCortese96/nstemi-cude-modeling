WITH filtered_labs AS (
    SELECT le.subject_id, le.hadm_id, le.itemid, le.charttime, le.valuenum, le.valueuom
    FROM `physionet-data.mimiciv_3_1_hosp.labevents` AS le
    JOIN (
        SELECT DISTINCT hadm_id
        FROM `physionet-data.mimiciv_3_1_hosp.diagnoses_icd`
        WHERE icd_code like '410%1' and not icd_code like '41071'
    ) AS diag
    ON le.hadm_id = diag.hadm_id
    WHERE le.itemid = 51003
)

SELECT subject_id, hadm_id, itemid, charttime, valuenum, valueuom
FROM filtered_labs
WHERE hadm_id IN (
    SELECT hadm_id
    FROM filtered_labs
    GROUP BY hadm_id
    HAVING COUNT(*) >= 4
)
ORDER BY hadm_id, charttime;
